```cpp 
class Notification
{
//WHy protected???Θα ήταν private αλλά επειδή έχω κλαη που κληρονομεί από το να κάθομαι να γράφω getter setter είναι να είναι protected.

    string MessageBody;

    ~Notification();
    //+Setters ,getters 
    virtual void send(){}; //Better bool 
    //2 choices:
    //1. Leave it blank(will never run but ok...)
    //2. MAKE IT SO THAT IT IS IMPLEMENTED BY SETTING IT TO 0!!! virtual void send() = 0;
    bool find();
};
class Email : public Notification
{
public:
Email(string mb, string mt, string re) : Notification(mb, mt, re){}
void send();
}
void Email::send()
{
    ...
    cout << "Email Sent" << endl;
}

int main()
{
    Email em("HELLO WORLD", "HELLO TITLE", "MY_RECIPINET");
    //Αν ήθελα να πω εγώ email θα έπρεπε να πω δώσε μου το όνομα του μηνύματος κλπ μετά....Και default constructor.
    //Τον default constructor τον φτιάνω στη notification!!!!
    //ΘΑ ΕΠΡΕΠΕ ΝΑ ΠΡΟΣΘΕΣΩ ΑΛΛΗ ΜΙΑ Notification
    em.send();

    Notification* pt = new Email(...);
    pt->send();
    //Both are the same!!!!
    pt->delete();
}

```
